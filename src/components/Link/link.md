
# Link

## Link component with similar functionality to anchor element

```js
import React from 'react';
import { Link } from '@platformbuilders/react-ui';

const Home: React.FC = () => {
   <Link onPress={() => {}}>
      Click here
   </Link>
  );
}

export default Home;
```

| ------------- | ------------- | ------------- |------------- |------------- |
